# react-native-styled-list

a styled list for react native

## Installation

```sh
npm install react-native-styled-list
```

## Usage

### `BulletedList`

```tsx
import { BulletedList } from "react-native-styled-list";

// ...

<BulletedList>
  <Text>item a</Text>
  <Text>item b</Text>
  <Text>item c</Text>
</BulletedList>
```

### `NumberedList`

```tsx
import { NumberedList } from "react-native-styled-list";

// ...

<NumberedList>
  <Text>item a</Text>
  <Text>item b</Text>
  <Text>item c</Text>
</NumberedList>
```
